struct imagine {
	double re;
	double im;
	imagine operator+ (const imagine& other) {
		return { re + other.re,im + other.im };
	}
	imagine operator* (const imagine& other) {
		return { re * other.re,im * other.im };
	}
	bool operator< (const imagine& other) {
		return { re * re + im * im < other.re* other.re + other.im * other.im };
	}
	bool operator> (const imagine& other) {
		return { re * re + im * im > other.re* other.re + other.im * other.im };
	}
};
